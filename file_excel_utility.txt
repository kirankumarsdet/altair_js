public class Utilities {

	FileInputStream fis;
	Workbook wb;

	public String fetchDataFromPropertyFile(String path, String key){
		fis = new FileInputStream(path);
		Properties p = new Properties();
		p.load(fis);
		String value = p.getProperty(key);
		return value;
	}

	public String getDataFromExcel(String path, String sheetName, int row, int col)
		{
		fis = new FileInputStream(path);
		wb = WorkbookFactory.create(fis);
		Sheet sheet1 = wb.getSheet(sheetName);
		String getData = sheet1.getRow(row).getCell(col).getStringCellValue();
		return getData;
	}

	public void sendDatatoExcel(String path, String sheetName, int row, int col, String val)
		{
		fis = new FileInputStream(path);
		wb = WorkbookFactory.create(fis);
		Sheet sheet1 = wb.getSheet(sheetName);
		Cell cell1 = sheet1.getRow(row).createCell(col);
		cell1.setCellValue(val);
		FileOutputStream fos = new FileOutputStream(path);
		wb.write(fos);
		wb.close();
	}

	public static void main(String[] args){
		Utilities ul = new Utilities();
		// fetchDataFrom Propertyfile
		String value = ul.fetchDataFromPropertyFile("./src/test/resources/js.Properties", "adminus");
		System.out.println(value);
		// fetchdatafrom excel
		String firstname = ul.getDataFromExcel("./src/test/resources/jsPractice.xlsx", "Sheet1", 1, 1);

		System.out.println(firstname);

		// sending data to excel

		ul.sendDatatoExcel("./src/test/resources/jsPractice.xlsx", "Sheet1", 3, 3, "friday");
	}

}
